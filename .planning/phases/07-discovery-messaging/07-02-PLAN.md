```typescript
import 'server-only';
import { getAdminFirestore } from '@/lib/firebase/admin';
import type { Conversation } from '@/lib/types/messages';

export function getConversationId(uid1: string, uid2: string): string {
  return [uid1, uid2].sort().join('_');
}

export async function getConversationsForUser(uid: string): Promise<Conversation[]> {
  const db = getAdminFirestore();
  const snap = await db.collection('conversations')
    .where('participantUids', 'array-contains', uid)
    .orderBy('lastMessageAt', 'desc')
    .limit(50)
    .get();
  return snap.docs.map(d => d.data() as Conversation);
}

export async function getMessages(conversationId: string, limitCount = 100) {
  const db = getAdminFirestore();
  const snap = await db.collection('conversations')
    .doc(conversationId)
    .collection('messages')
    .orderBy('createdAt', 'asc')
    .limit(limitCount)
    .get();
  return snap.docs.map(d => ({ messageId: d.id, ...d.data() }));
}
```

- Compute `conversationId = getConversationId(currentUid, otherUid)`
- Check if conversation doc exists with `db.collection('conversations').doc(conversationId).get()`
- If not exists: fetch both userProfiles docs, extract `{ displayName, avatarUrl: avatarUrl || null, handle }` for each
- Create conversation doc with: `conversationId`, `participantUids: [currentUid, otherUid]`, `participantProfiles` map, `lastMessage: ''`, `lastMessageAt: FieldValue.serverTimestamp()`, `lastMessageSenderUid: ''`, `unreadCounts: { [currentUid]: 0, [otherUid]: 0 }`, `createdAt: FieldValue.serverTimestamp()`
- Return `conversationId`

**`sendMessage(conversationId: string, text: string): Promise<void>`**
- Get current user uid from getTokens
- Guard: text must be non-empty after trim, max 2000 chars
- Fetch current user's profile for displayName and avatarUrl
- Write message doc to `conversations/{conversationId}/messages` with auto-id: `{ senderUid, senderDisplayName, senderAvatarUrl, text: text.trim(), createdAt: FieldValue.serverTimestamp(), seenAt: null, seenBy: [] }`
- Update conversation doc: `{ lastMessage: text.trim().slice(0, 80), lastMessageAt: FieldValue.serverTimestamp(), lastMessageSenderUid: currentUid }`, and increment `unreadCounts.{otherUid}` by 1 using `FieldValue.increment(1)`. Get otherUid from conversation's `participantUids` array (the one that is not currentUid).

**`markConversationRead(conversationId: string): Promise<void>`**
- Get current user uid
- Update conversation doc: set `unreadCounts.{currentUid}` to 0
- Query messages in the conversation where `senderUid != currentUid` and `seenAt == null`, update each with `{ seenAt: FieldValue.serverTimestamp(), seenBy: FieldValue.arrayUnion(currentUid) }`

**`updateLastSeen(): Promise<void>`**
- Get current user uid from getTokens
- If no user, return silently (guest)
- Update `userProfiles/{uid}` with `{ lastSeen: FieldValue.serverTimestamp() }` using `{ merge: true }`
