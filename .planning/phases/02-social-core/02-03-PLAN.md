**1. Create `src/lib/firestore/posts.ts`** — query helpers and utilities:

```typescript
import { getAdminFirestore } from '@/lib/firebase/admin';
import type { Post } from '@/lib/types/social';

export function buildSearchKeywords(text: string): string[] {
  return [...new Set(
    text.toLowerCase()
      .split(/\s+|[^\w]/)
      .filter(w => w.length >= 3)
  )];
}

export async function getPost(postId: string): Promise<Post | null> {
  const db = getAdminFirestore();
  const doc = await db.collection('posts').doc(postId).get();
  return doc.exists ? (doc.data() as Post) : null;
}
```

**2. Create `src/app/actions/posts.ts`** — `'use server'` directive at top:

**`createPost(uid: string, data: { text: string; imageUrl: string | null; category: string; commentsRestricted: 'all' | 'followers'; linkPreview: LinkPreview | null })`:**
- Validate: text 1-5000 chars, category must be in `ORTHODOX_CATEGORIES`
- Fetch author profile from `userProfiles/{uid}` to get denormalized fields (handle, displayName, avatarUrl, jurisdictionId)
- Get author roleLevel from Auth custom claims via `getAdminAuth().getUser(uid)` then `.customClaims.roleLevel`
- Generate `searchKeywords` via `buildSearchKeywords(data.text)`
- Create post doc in `posts/` collection with all fields from the Post interface
- Fan-out: query `follows` where `followedUid == uid`, get all `followerUid` values, add `uid` itself to the list
- Batch write the post to each follower's `users/{followerUid}/userFeed/{postId}` (chunk into 500-operation batches per Firestore limit)
- Increment `postCount` on `userProfiles/{uid}` via `FieldValue.increment(1)`
- Return `{ success: true, postId }` or `{ success: false, error: string }`

**`deletePost(uid: string, postId: string)`:**
- Verify post exists and `authorUid === uid` (or caller is moderator+)
- Delete the post document from `posts/{postId}`
- Delete all docs in `posts/{postId}/likes` subcollection (list all, batch delete)
- Delete all docs in `posts/{postId}/comments` subcollection (list all, batch delete)
- Query all `follows` where `followedUid == uid`, get follower UIDs + own UID
- Remove `users/{followerUid}/userFeed/{postId}` for each follower (batch delete)
- Decrement `postCount` on `userProfiles/{uid}` via `FieldValue.increment(-1)`
- Return `{ success: true }` or `{ success: false, error: string }`

**`editPost(uid: string, postId: string, data: { text: string; category: string; commentsRestricted: 'all' | 'followers' })`:**
- Verify post exists and `authorUid === uid`
- Update post doc: `text`, `category`, `commentsRestricted`, `searchKeywords` (rebuilt), `updatedAt: FieldValue.serverTimestamp()`, `isEdited: true`
- Update feed entries: query `follows` where `followedUid == uid`, update `users/{uid}/userFeed/{postId}` for each follower + self with same fields
- Return `{ success: true }` or `{ success: false, error: string }`

**3. Create `src/app/actions/linkPreview.ts`** — `'use server'`:

Install `open-graph-scraper` first: `npm install open-graph-scraper`

```typescript
'use server';
import ogs from 'open-graph-scraper';
import type { LinkPreview } from '@/lib/types/social';

export async function fetchLinkPreview(url: string): Promise<LinkPreview | null> {
  try {
    const { result } = await ogs({ url, timeout: 5000 });
    if (!result.success) return null;
    return {
      url,
      title: result.ogTitle ?? null,
      description: result.ogDescription ?? null,
      imageUrl: result.ogImage?.[0]?.url ?? null,
      siteName: result.ogSiteName ?? null,
    };
  } catch {
    return null;
  }
}
```
