**5. Create `src/app/(main)/calendar/page.tsx`:**

Server Component. Reads `searchParams.date` (optional YYYY-MM-DD) and `searchParams.cal` (optional calendar override for guests).

```typescript
import { Suspense } from 'react';
import { cookies } from 'next/headers';
import { getTokens } from 'next-firebase-auth-edge';
import { fetchDayData } from '@/lib/calendar/orthocal';
import { getProfileByUid } from '@/lib/firestore/profiles';
import { CalendarDayView } from '@/components/calendar/CalendarDayView';
import { CalendarSkeleton } from '@/components/calendar/CalendarSkeleton';
import type { CalendarSystem } from '@/lib/types/calendar';

// authConfig pattern — inline as in existing pages
const authConfig = { /* same pattern as profile/edit/page.tsx */ };

export default async function CalendarPage({
  searchParams,
}: {
  searchParams: Promise<{ date?: string; cal?: string }>;
}) {
  const params = await searchParams;

  // Determine date
  const civilDate = params.date ? new Date(params.date + 'T12:00:00') : new Date();

  // Determine calendar preference
  let calendar: CalendarSystem = 'new_julian'; // default for guests
  let isLoggedIn = false;

  try {
    const tokens = await getTokens(await cookies(), authConfig);
    if (tokens) {
      isLoggedIn = true;
      const profile = await getProfileByUid(tokens.decodedToken.uid);
      if (profile?.calendarPreference) {
        calendar = profile.calendarPreference as CalendarSystem;
      }
    }
  } catch {
    // Guest user — use default
  }

  // Allow URL override (for guest toggling)
  if (params.cal === 'old_julian' || params.cal === 'new_julian') {
    calendar = params.cal;
  }

  // Fetch day data
  const dayData = await fetchDayData(civilDate, calendar);

  return (
    <div className="max-w-2xl mx-auto px-4 py-8">
      <h1 className="font-cinzel text-gold text-xl uppercase tracking-widest mb-6">
        Liturgical Calendar
      </h1>
      <Suspense fallback={<CalendarSkeleton />}>
        <CalendarDayView
          initialDay={dayData}
          initialCalendar={calendar}
          isLoggedIn={isLoggedIn}
        />
      </Suspense>
    </div>
  );
}
```
