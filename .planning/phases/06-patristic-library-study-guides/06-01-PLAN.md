```typescript
import type { OrthodoxCategory } from '@/lib/constants/categories';

export type PatristicEra = 'apostolic' | 'ante-nicene' | 'nicene' | 'post-nicene';

export interface PatristicText {
  textId: string;            // e.g., "ignatius-ephesians" or "chrysostom-homily-matthew-1"
  authorSlug: string;        // FK to patristic_authors, e.g., "ignatius-of-antioch"
  authorName: string;        // Denormalized: "Ignatius of Antioch"
  era: PatristicEra;
  title: string;             // e.g., "Epistle to the Ephesians"
  workTitle: string;         // Parent work title if chapter: "Homilies on Matthew"
  chapterOrHomily: number | null; // null for standalone works, 1-N for chapters
  topics: OrthodoxCategory[];
  source: string;            // e.g., "ANF Vol. 1, Philip Schaff ed. (CCEL)"
  sortOrder: number;
  body: string;              // Full text stripped of XML markup
  searchKeywords: string[];  // title + authorName + topics + ~100 significant body words
}

export interface PatristicAuthor {
  authorSlug: string;        // doc ID, e.g., "ignatius-of-antioch"
  name: string;
  era: PatristicEra;
  eraLabel: string;          // Display: "Apostolic Father (c. 35-108 AD)"
  feastDay: string | null;   // e.g., "December 20 (OC)"
  keyContribution: string;   // One-sentence summary
  bio: string;               // 2-4 sentence biography
  sortOrder: number;
}

export interface StudyGuideItem {
  step: number;
  title: string;
  description: string;
  type: 'patristic' | 'scripture';
  refId: string | null;      // patristic_texts textId OR null for scripture
  href: string | null;       // null for patristic; "/scripture/john/1#verse-1" for scripture
}

export interface StudyGuide {
  guideId: string;           // doc ID, e.g., "introduction-to-orthodoxy"
  slug: string;              // URL slug (same as guideId)
  title: string;
  description: string;
  topic: OrthodoxCategory;
  items: StudyGuideItem[];
}
```

**XML Parser configuration:**
```typescript
const parser = new XMLParser({
  ignoreAttributes: false,
  attributeNamePrefix: '@_',
  textNodeName: '#text',
  trimValues: true,
  ignoreDeclaration: true,
  removeNSPrefix: true,
});
```

**Core functions:**

`extractBodyText(node: unknown): string` -- Recursively collect text from an XML node tree. Skip nodes with tag names: `note`, `index`, `scripRef`, `pb`. For arrays, recurse each element. For objects with `#text`, collect the text. For objects with `p` children, recurse into them. Return concatenated plain text with double-space-collapsed whitespace.

`buildPatristicKeywords(title, authorName, topics, body, maxBodyWords=100)` -- Inline version (same logic as the one in `src/lib/firestore/patristic.ts` but inlined to avoid Next.js module resolution outside `src/`). This is the established pattern from Phase 5 where seed scripts inline their keyword builder.

**Seed data approach -- use HARDCODED patristic text data rather than live XML downloads.** The CCEL XML files are large (2-10MB each), structurally inconsistent across volumes, and downloading them at seed time introduces network dependency and parsing fragility. Instead:

- Define an array `SEED_TEXTS` of 20+ hardcoded PatristicText objects with pre-written `body` text (representative excerpts, 200-500 words each). This mirrors how the EOB NT seed used pre-parsed JSON input.
- Define an array `SEED_AUTHORS` of 10-12 PatristicAuthor objects with handwritten bios (canonically accurate, jurisdiction-neutral, 2-4 sentences each).
- Define an array `SEED_GUIDES` of 4 StudyGuide objects.

**Why hardcoded:** Research notes the CCEL ThML structure varies across volumes (Pitfall 3), footnote pollution risk (Pitfall 1), and structural ambiguity. For 20 seed documents the hardcoded approach is faster, more reliable, and produces canonically accurate content. The extractBodyText + XMLParser functions should still be included in the script (commented out with a usage example) for future bulk ingestion.
